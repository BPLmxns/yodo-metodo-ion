  <script>
    let muestrasPendientes = [];

    window.onload = () => {
      google.script.run
        .withSuccessHandler(res => {
          muestrasPendientes = res;
          renderizarTabla(muestrasPendientes);
        })
        .withFailureHandler(err => { alert("Error: " + err.message); document.getElementById('loader').style.display='none'; })
        .obtenerDatosInformes();
    };

    function mostrarToastError(msg) {
      const t = document.getElementById('toast-container');
      const d = document.createElement('div');
      d.className = `toast toast-error`;
      d.innerHTML = `⚠️ ${msg}`;
      t.appendChild(d);
      setTimeout(() => d.classList.add('show'), 10);
      setTimeout(() => { d.classList.remove('show'); setTimeout(() => d.remove(), 300); }, 3500);
    }

    function renderizarTabla(muestras) {
      const tbody = document.getElementById('tbodyInformes');
      tbody.innerHTML = '';

      if (muestras.length === 0) {
        tbody.innerHTML = `<tr><td colspan="6" style="padding:30px; color:var(--gris); font-style:italic;">No hay muestras pendientes de reporte.</td></tr>`;
        document.getElementById('loader').style.display = 'none';
        return;
      }

      muestras.forEach((m, idx) => {
        const iconAC = m.ac ? `<span class="ac-icon ac-check">✔</span>` : `<span class="ac-icon ac-cross">✖</span>`;
        
        let chkHTML = "";
        let rowClass = "";

        if (m.enviado) {
          rowClass = "row-enviada";
          chkHTML = `<span class="badge bg-enviada">ENVIADO</span>`;
        } else {
          const disabledAttr = m.ac ? '' : 'disabled';
          chkHTML = `<input type="checkbox" class="chk-muestra" value="${m.control}" onchange="evaluarBoton()" ${disabledAttr}>`;
        }
        
        let historialHTML = '';
        if (m.historial.length === 0) {
          historialHTML = `<tr><td colspan="9" style="color:var(--gris); font-style:italic;">No se encontraron registros en el Control de Duplicados.</td></tr>`;
        } else {
          m.historial.forEach(h => {
            const badgeClass = String(h.revision).toUpperCase().includes("NO CUMPLE") ? 'bg-nocumple' : 'bg-cumple';
            historialHTML += `
              <tr>
                <td><strong>${h.tipo}</strong></td>
                <td>${h.repA}</td>
                <td>${h.resA}</td>
                <td>${h.repB}</td>
                <td>${h.resB}</td>
                <td style="color:var(--gris);">${h.unidad}</td>
                <td><strong style="color:var(--azul-oscuro);">${h.promedio}</strong></td>
                <td>${h.horrat}</td>
                <td><span class="badge ${badgeClass}">${h.revision}</span></td>
              </tr>
            `;
          });
        }

        tbody.innerHTML += `
          <tr class="${rowClass}" id="fila_main_${idx}">
            <td><strong id="lbl_ctrl_${idx}">${m.control}</strong></td>
            <td class="text-left"><span style="font-size:11px; display:block; line-height:1.2;">${m.producto}</span></td>
            <td><strong>${m.maxRep}</strong></td>
            <td>${iconAC}</td>
            <td><button class="btn-detalles" onclick="toggleDetalles(${idx})">Ver detalles ▼</button></td>
            <td id="celda_chk_${idx}">${chkHTML}</td>
          </tr>
          <tr id="det_${idx}" class="row-detalles">
            <td colspan="6" style="padding: 0;">
              <table class="tabla-secundaria">
                <thead>
                  <tr>
                    <th>Tipo de Muestra</th>
                    <th>Rep A</th>
                    <th>Resultado A</th>
                    <th>Rep B</th>
                    <th>Resultado B</th>
                    <th>Unidad</th>
                    <th>Promedio</th>
                    <th>HorRat(r)</th>
                    <th>Estado / Revisión</th>
                  </tr>
                </thead>
                <tbody>${historialHTML}</tbody>
              </table>
            </td>
          </tr>
        `;
      });

      document.getElementById('loader').style.display = 'none';
      evaluarBoton();
    }

    function toggleDetalles(idx) {
      const fila = document.getElementById(`det_${idx}`);
      if (fila.style.display === 'table-row') {
        fila.style.display = 'none';
      } else {
        fila.style.display = 'table-row';
      }
    }

    function evaluarBoton() {
      const checkboxes = document.querySelectorAll('.chk-muestra:checked');
      const btn = document.getElementById('btnSolicitar');
      btn.disabled = checkboxes.length === 0;
    }

    function solicitarRevision() {
      const checkboxes = document.querySelectorAll('.chk-muestra:checked');
      const controlesSeleccionados = Array.from(checkboxes).map(chk => chk.value);
      
      const btn = document.getElementById('btnSolicitar');
      btn.disabled = true;

      // Encender Pantalla de Carga Wizard
      document.getElementById('overlay-submit').style.display = 'flex';
      document.getElementById('submit-spinner').style.display = 'block';
      document.getElementById('submit-icon-success').style.display = 'none';
      document.getElementById('submit-title').innerText = 'Enviando solicitud...';
      document.getElementById('submit-title').style.color = 'var(--azul-oscuro)';
      document.getElementById('submit-text').innerText = 'Guardando datos y notificando al líder.';
      
      google.script.run
        .withSuccessHandler(res => {
          // Marcar las muestras locales como enviadas para que si no se cierra de golpe, se bloqueen
          muestrasPendientes.forEach(m => {
            if (controlesSeleccionados.includes(m.control)) m.enviado = true;
          });
          renderizarTabla(muestrasPendientes);

          // Transición a Éxito
          document.getElementById('submit-spinner').style.display = 'none';
          document.getElementById('submit-icon-success').style.display = 'block';
          document.getElementById('submit-title').innerText = '¡Solicitud Enviada!';
          document.getElementById('submit-title').style.color = 'var(--success)';
          document.getElementById('submit-text').innerText = res + '\nCerrando ventana...';
          
          setTimeout(() => google.script.host.close(), 2500);
        })
        .withFailureHandler(err => {
          document.getElementById('overlay-submit').style.display = 'none';
          mostrarToastError(err.message);
          evaluarBoton();
        })
        .enviarSolicitudRevision(controlesSeleccionados);
    }
  </script>
