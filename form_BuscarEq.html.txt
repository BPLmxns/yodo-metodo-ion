  </div>

  <script>
    let DATOS_EQUIPOS = [];

    window.onload = () => {
      google.script.run
        .withSuccessHandler(res => {
          DATOS_EQUIPOS = res.equipos;
          
          const ops = res.areas.map(a => `<option value="${a}">${a}</option>`).join('');
          document.getElementById('filtroArea').innerHTML = '<option value="">-- Todas las áreas --</option>' + ops;
          
          renderTabla();
          document.getElementById('loader').style.display = 'none';
        })
        .withFailureHandler(err => {
          document.getElementById('loader').style.display = 'none';
          mostrarToastError(err.message);
        })
        .obtenerDatosEquipos();
    };

    function mostrarToastError(msg) {
      const t = document.getElementById('toast-container');
      const d = document.createElement('div');
      d.className = `toast toast-error`;
      d.innerHTML = `⚠️ ${msg}`;
      t.appendChild(d);
      setTimeout(() => d.classList.add('show'), 10);
      setTimeout(() => { d.classList.remove('show'); setTimeout(() => d.remove(), 300); }, 3500);
    }

    function toggleClearBtn() {
      const val = document.getElementById('buscadorTxt').value;
      document.getElementById('clearBtn').style.display = val ? 'block' : 'none';
    }

    function limpiarFiltro() {
      const input = document.getElementById('buscadorTxt');
      input.value = '';
      input.focus();
      toggleClearBtn();
      renderTabla();
    }

    function renderTabla() {
      const areaFiltro = document.getElementById('filtroArea').value;
      const textoFiltro = document.getElementById('buscadorTxt').value.toUpperCase().trim();
      const body = document.getElementById('bTable');
      const cont = document.getElementById('contador');
      
      body.innerHTML = "";

      const filtrados = DATOS_EQUIPOS.filter(eq => {
        let pasaArea = !areaFiltro || eq.area === areaFiltro;
        let pasaTexto = !textoFiltro || eq.codigo.toUpperCase().includes(textoFiltro);
        return pasaArea && pasaTexto;
      });
      
      if (filtrados.length === 0) {
        body.innerHTML = `<tr><td colspan="2" style="text-align:center; padding:30px; color:var(--gris); font-style:italic;">No se encontraron equipos que coincidan con la búsqueda.</td></tr>`;
        cont.innerText = `Total: 0 equipos`;
        return;
      }

      body.innerHTML = filtrados.map(eq => `
        <tr>
          <td><strong style="color:var(--azul-oscuro); font-size:13px;">${eq.codigo}</strong></td>
          <td>${eq.area || '<span style="color:var(--gris); font-style:italic;">No especificada</span>'}</td>
        </tr>
      `).join('');
      
      cont.innerText = `Total: ${filtrados.length} equipos`;
    }
  </script>
