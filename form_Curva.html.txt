    function mostrarToastError(msg) {
      const t = document.getElementById('toast-container');
      const d = document.createElement('div');
      d.className = 'toast toast-error';
      d.innerHTML = `⚠️ ${msg}`;
      t.appendChild(d);
      setTimeout(() => d.classList.add('show'), 10);
      setTimeout(() => { d.classList.remove('show'); setTimeout(() => d.remove(), 300); }, 3500);

    function inicializarFormulario(datos) {
      dataMicropipetas = datos.micropipetas;
      dataPipetas = datos.pipetas;
      dataPipetas5mL = datos.pipetas5mL; 
      dataBalones = datos.balones;

    function cambiarTipoInst(id) {
      const inputEl = document.getElementById(`inst_${id}`);
      if(inputEl) {
        inputEl.value = ''; 
        document.getElementById(`clear_inst_${id}`).style.display = 'none';
      }
      actualizarDropdown(id);
    }

    function actualizarDropdown(id) {
      const selector = document.querySelector(`input[name="tipo_${id}"]:checked`);
      if(!selector) return;
      let arr = (selector.value === 'micro') ? dataMicropipetas : ((id === 'Int') ? dataPipetas5mL : dataPipetas);
      poblarListaCombo(arr, `lista_${id}`, `inst_${id}`);
    }

    function guardarEstadoTabla() {
      niveles.forEach((nivel, idx) => {
        if (idx !== 0) {
          const radioSel = document.querySelector(`input[name="tipo_${idx}"]:checked`);
          if (radioSel) nivel.tipoInst = radioSel.value;
          
          const inputInst = document.getElementById(`inst_${idx}`);
          if (inputInst) nivel.inst = inputInst.value;
        }
        
        const inputBalon = document.getElementById(`balon_${idx}`);
        if (inputBalon) nivel.balon = inputBalon.value;
      });
    }

        const isFijo = (nivel.tipo === 'fijo');
        
        let classAli = isFijo ? 'td-input read-only' : 'td-input';
        let readOnlyAttr = isFijo ? 'readonly' : '';
        if (nivel.tipo === 'extra') {
          const prevBound = niveles[idx - 1] ? niveles[idx - 1].alicuota : 0;
          const nextBound = niveles[idx + 1] ? niveles[idx + 1].alicuota : Infinity;
          if (nivel.alicuota <= prevBound || nivel.alicuota >= nextBound) classAli += ' invalid';
        }
        
        let botones = `<button class="btn-icon" onclick="agregarNivel(${idx})" title="Insertar punto debajo">+</button>`;
        if (!isFijo) botones += `<button class="btn-icon btn-icon-red" onclick="quitarNivel(${idx})" title="Eliminar punto">-</button>`;

        let instHTML = (idx === 0) ? `<input type="text" class="td-input read-only" readonly value="N/A">` : 
          `<div style="display:flex; justify-content:center; gap:8px; margin-bottom:4px;">
            <label class="radio-label"><input type="radio" name="tipo_${idx}" value="micro" onchange="cambiarTipoInst('${idx}')" ${nivel.tipoInst === 'micro' ? 'checked' : ''}> Micro</label>
            <label class="radio-label"><input type="radio" name="tipo_${idx}" value="pipeta" onchange="cambiarTipoInst('${idx}')" ${nivel.tipoInst === 'pipeta' ? 'checked' : ''}> Pipeta</label>
          </div>
          <div class="dropdown-container">
            <input type="text" id="inst_${idx}" class="td-input" placeholder="Buscar..." autocomplete="off" onfocus="mostrarOpciones('lista_${idx}', 'inst_${idx}')" oninput="filtrarOpciones('lista_${idx}', 'inst_${idx}')" value="${nivel.inst}">
            <span class="clear-btn" id="clear_inst_${idx}" onclick="limpiarDropdown('lista_${idx}', 'inst_${idx}')" style="${nivel.inst ? 'display:block' : 'display:none'}">✖</span>
            <span class="dropdown-chevron">▼</span><div class="dropdown-list" id="lista_${idx}"></div>
          </div>`;

        let balonHTML = `<div class="dropdown-container">
            <input type="text" id="balon_${idx}" class="td-input" placeholder="Buscar..." autocomplete="off" onfocus="mostrarOpciones('lista_balon_${idx}', 'balon_${idx}')" oninput="filtrarOpciones('lista_balon_${idx}', 'balon_${idx}')" value="${nivel.balon}">
            <span class="clear-btn" id="clear_balon_${idx}" onclick="limpiarDropdown('lista_balon_${idx}', 'balon_${idx}')" style="${nivel.balon ? 'display:block' : 'display:none'}">✖</span>
            <span class="dropdown-chevron">▼</span><div class="dropdown-list" id="lista_balon_${idx}"></div>
          </div>`;

        tbody.innerHTML += `
          <tr>
            <td><strong>${nivel.nombre}</strong></td>
            <td>${instHTML}</td><td>${balonHTML}</td>
            <td><input type="number" id="ali_${idx}" class="${classAli}" ${readOnlyAttr} value="${nivel.alicuota}" step="0.1" oninput="actualizarAlicuota(${idx}, this.value)"></td>
            <td><strong id="conc_${idx}" style="color:var(--azul-oscuro);">-</strong></td>
            <td><span id="log_${idx}" style="color:var(--gris); font-family:monospace;">-</span></td>
            <td>${botones}</td>
          </tr>
        `;
      });
      recalcularCurva();
      
      if (dataBalones.length > 0) {
        niveles.forEach((_, idx) => {
          if (idx !== 0) actualizarDropdown(idx); 
          poblarListaCombo(dataBalones, `lista_balon_${idx}`, `balon_${idx}`);
        });
      }
    }

    function actualizarAlicuota(idx, val) {
      const numVal = parseFloat(val);
      niveles[idx].alicuota = isNaN(numVal) ? 0 : numVal;

      if (niveles[idx].tipo === 'extra') {
        const inputEl = document.getElementById(`ali_${idx}`);
        const prevBound = niveles[idx - 1] ? niveles[idx - 1].alicuota : 0;
        const nextBound = niveles[idx + 1] ? niveles[idx + 1].alicuota : Infinity;
        if (niveles[idx].alicuota <= prevBound || niveles[idx].alicuota >= nextBound) inputEl.classList.add('invalid');
        else inputEl.classList.remove('invalid');
      }
      recalcularCurva();
    }

    function agregarNivel(idx) {
      guardarEstadoTabla(); 
      const prevVal = niveles[idx].alicuota;
      const nextVal = niveles[idx + 1] ? niveles[idx + 1].alicuota : prevVal + 1.0;
      const defaultAlicuota = parseFloat(((prevVal + nextVal) / 2).toFixed(2));
      niveles.splice(idx + 1, 0, { tipo: 'extra', nombre: 'Extra', alicuota: defaultAlicuota, conc: 0, log: 0, mv: "", tipoInst: "micro", inst: "", balon: "" });
      renderizarTabla();
    }
    
    function quitarNivel(idx) { 
      guardarEstadoTabla(); 
      niveles.splice(idx, 1); 
      renderizarTabla(); 
    }

    function cargarDetallesLote() {
      const lote = document.getElementById('loteYoduro').value;
      if (!lote) { limpiarCalculos(); return; }
      document.getElementById('mini-loader').style.display = 'flex';
      
      google.script.run
        .withSuccessHandler(res => {
          document.getElementById('pesoKi').value = res.pesoKI;
          document.getElementById('balanza').value = res.balanzaKI;
          document.getElementById('balonStd').value = res.balonInstrumental;
          document.getElementById('volStdVal').value = res.volumenAgua; 
          recalcularCurva(); 
          document.getElementById('mini-loader').style.display = 'none';
        }).withFailureHandler(err => { 
          mostrarToastError(err.message); 
          limpiarCalculos(); 
          document.getElementById('mini-loader').style.display = 'none'; 
        })
        .obtenerDetallesLote(lote);
    }

    function recalcularCurva() {
      const pesoMg = parseFloat(document.getElementById('pesoKi').value) || 0;
      const volStdMl = parseFloat(document.getElementById('volStdVal').value) || 100;
      
      let concMadre = 0, concInt = 0;
      if (pesoMg > 0) {
        concMadre = (pesoMg * FACTOR_KI * 1000) / volStdMl;
        document.getElementById('concStd').value = concMadre.toFixed(1);
        concInt = (concMadre * 5.0) / 100; 
        document.getElementById('concInt').value = concInt.toFixed(2);
      } else {
        document.getElementById('concStd').value = ""; document.getElementById('concInt').value = "";
      }

      niveles.forEach((nivel, idx) => {
        nivel.conc = 0; nivel.log = "N/A";
        if (nivel.alicuota > 0 && concInt > 0) {
          nivel.conc = (concInt * nivel.alicuota) / VOLUMEN_FINAL_CURVA;
          nivel.log = Math.log10(nivel.conc).toFixed(4);
        }
        const tdConc = document.getElementById(`conc_${idx}`);
        const tdLog = document.getElementById(`log_${idx}`);
        if(tdConc) tdConc.innerText = nivel.conc > 0 ? nivel.conc.toFixed(3) : "-";
        if(tdLog) tdLog.innerText = nivel.log;
      });
    }

    function limpiarCalculos() {
      document.getElementById('pesoKi').value = ""; 
      document.getElementById('balanza').value = "";
      document.getElementById('balonStd').value = ""; 
      document.getElementById('concStd').value = ""; 
      document.getElementById('concInt').value = "";
      niveles.forEach((_, idx) => {
        const tdConc = document.getElementById(`conc_${idx}`), tdLog = document.getElementById(`log_${idx}`);
        if(tdConc) tdConc.innerText = "-"; if(tdLog) tdLog.innerText = "-";
      });
    }

    function poblarListaCombo(arr, listId, inputId) {
      const lista = document.getElementById(listId); lista.innerHTML = '';
      if(arr.length === 0) { lista.innerHTML = '<div class="no-results">Sin datos</div>'; return; }
      arr.forEach(item => {
        let div = document.createElement('div'); div.className = 'dropdown-option'; div.innerText = item;
        div.onclick = () => { 
          document.getElementById(inputId).value = item; 
          document.getElementById(inputId).classList.remove('invalid'); 
          lista.classList.remove('show');
          const clearBtn = document.getElementById('clear_' + inputId);
          if (clearBtn) clearBtn.style.display = 'block';
        };
        lista.appendChild(div);
      });
    }

    function mostrarOpciones(listId, inputId) { 
      document.getElementById(listId).classList.add('show'); 
      filtrarOpciones(listId, inputId); 
    }

    function filtrarOpciones(listId, inputId) {
      const inputVal = document.getElementById(inputId).value.toUpperCase();
      const lista = document.getElementById(listId);
      const opciones = lista.getElementsByClassName('dropdown-option');
      let visibles = 0;
      
      for (let i = 0; i < opciones.length; i++) {
        if (opciones[i].innerText.toUpperCase().indexOf(inputVal) > -1) { 
          opciones[i].style.display = ""; visibles++; 
        } else { 
          opciones[i].style.display = "none"; 
        }
      }
      
      let noRes = lista.querySelector('.no-results');
      if (visibles === 0 && opciones.length > 0) { 
        if (!noRes) { noRes = document.createElement('div'); noRes.className = 'no-results'; noRes.innerText = 'Sin coincidencias'; lista.appendChild(noRes); }
      } else if(noRes && visibles > 0) { noRes.remove(); }

      // Mostrar u ocultar la X
      const clearBtn = document.getElementById('clear_' + inputId);
      if(clearBtn) clearBtn.style.display = inputVal ? 'block' : 'none';

      // ================= CÁLCULO INTELIGENTE DE POSICIÓN (Arriba / Abajo) =================
      const rect = document.getElementById(inputId).getBoundingClientRect();
      const windowHeight = window.innerHeight;
      
      if (rect.bottom + 160 > windowHeight && rect.top > 160) {
        // Se sale por abajo, así que lo lanzamos hacia arriba
        lista.style.top = 'auto';
        lista.style.bottom = '100%';
        lista.style.marginTop = '0';
        lista.style.marginBottom = '4px';
        lista.style.boxShadow = '0 -4px 12px rgba(0,0,0,0.15)';
      } else {
        // Despliegue normal hacia abajo
        lista.style.top = '100%';
        lista.style.bottom = 'auto';
        lista.style.marginTop = '4px';
        lista.style.marginBottom = '0';
        lista.style.boxShadow = '0 4px 12px rgba(0,0,0,0.15)';
      }
    }

    function limpiarDropdown(listId, inputId) {
      const input = document.getElementById(inputId);
      input.value = '';
      input.focus();
      filtrarOpciones(listId, inputId);
    }

    function calcularRegresion() {
      let mvBlanco = parseFloat(niveles[0].mv);
      if (isNaN(mvBlanco)) mvBlanco = 0; 

      let xData = [];
      let yData = []; 

      niveles.forEach((n, idx) => {
        let yCorrTexto = "-";
        if (n.mv !== "") {
          const valY = parseFloat(n.mv);
          if (!isNaN(valY)) {
            const corregido = valY - mvBlanco;
            yCorrTexto = corregido.toFixed(1);
            
            if (n.log !== "N/A") {
              const x = parseFloat(n.log);
              if (!isNaN(x)) {
                xData.push(x);
                yData.push(corregido); 
              }
            }
          }
        }
        
        const tdCorr = document.getElementById(`corr_${idx}`);
        if(tdCorr) tdCorr.innerText = yCorrTexto;
      });

      const nDatos = xData.length;
      let a = 0, b = 0, r2 = 0;

      if (nDatos >= 3) { 
        let sumX = 0, sumY = 0, sumXY = 0, sumXX = 0, sumYY = 0;
        for (let i = 0; i < nDatos; i++) {
          sumX += xData[i]; sumY += yData[i];
          sumXY += xData[i] * yData[i];
          sumXX += xData[i] * xData[i];
          sumYY += yData[i] * yData[i];
        }

        const xMean = sumX / nDatos;
        const yMean = sumY / nDatos;

        const num = sumXY - (sumX * sumY / nDatos);
        const denX = sumXX - (sumX * sumX / nDatos);
        const denY = sumYY - (sumY * sumY / nDatos);

        a = num / denX;
        b = yMean - (a * xMean);
        r2 = (num * num) / (denX * denY);

        document.getElementById('res_a').innerText = a.toFixed(2);
        document.getElementById('res_b').innerText = b.toFixed(2);
        document.getElementById('res_r2').innerText = r2.toFixed(4); 
        
        let signoB = b >= 0 ? "+" : "-";
        document.getElementById('ecuacion').innerText = `y = ${a.toFixed(2)}x ${signoB} ${Math.abs(b).toFixed(2)}`;

        const badge = document.getElementById('badgeCurva');
        if (r2 >= 0.990) {
          badge.innerText = "CUMPLE";
          badge.className = "badge badge-conforme"; 
        } else {
          badge.innerText = "NO CUMPLE";
          badge.className = "badge badge-noconforme"; 
        }

        dibujarGrafica(xData, yData, a, b);

      } else {
        document.getElementById('res_a').innerText = "-";
        document.getElementById('res_b').innerText = "-";
        document.getElementById('res_r2').innerText = "-";
        document.getElementById('ecuacion').innerText = "Faltan datos de mV";
        const badge = document.getElementById('badgeCurva');
        badge.innerText = "Pendiente"; badge.className = "badge";
        
        if (chartInstance) chartInstance.destroy();
      }
    }

    function dibujarGrafica(xData, yData, a, b) {
      if (chartInstance) chartInstance.destroy();

      const xMin = Math.min(...xData);
      const xMax = Math.max(...xData);
      
      const lineData = [
        {x: xMin, y: (a * xMin + b)},
        {x: xMax, y: (a * xMax + b)}
      ];

      const scatterData = xData.map((x, i) => ({ x: x, y: yData[i] }));

      const ctx = document.getElementById('graficaCurva').getContext('2d');
      chartInstance = new Chart(ctx, {
        type: 'scatter',
        data: {
          datasets: [{
            label: 'Lectura Corregida',
            data: scatterData,
            backgroundColor: '#0a4079',
            pointRadius: 6,
            order: 1
          }, {
            label: 'Tendencia',
            data: lineData,
            type: 'line',
            borderColor: '#afcc46',
            borderWidth: 2,
            fill: false,
            pointRadius: 0,
            order: 2
          }]
        },
        options: {
          responsive: true,
          maintainAspectRatio: false,
          plugins: { legend: { display: false } },
          scales: {
            x: { title: { display: true, text: 'Log(Concentración)' } },
            y: { title: { display: true, text: 'Lectura Corregida [mV]' } }
          }
        }
      });
    }

    function finalizarCurva() {
      const r2Text = document.getElementById('res_r2').innerText;
      if (r2Text === "-" || r2Text === "") {
        mostrarToastError("Faltan datos de lecturas (mV) para poder guardar la curva.");
        return;
      }

      const instIntVal = document.getElementById('inst_Int').value;
      const balonInt = document.getElementById('balonInt').value;
      if (!instIntVal || !balonInt) {
        mostrarToastError("Asegúrese de haber seleccionado el instrumento y balón en el Segmento I.");
        return;
      }

      const datos = {
        balanza: document.getElementById('balanza').value,
        loteYoduro: document.getElementById('loteYoduro').value,
        pesoKi: document.getElementById('pesoKi').value,
        balonStd: document.getElementById('balonStd').value,
        concStd: document.getElementById('concStd').value,
        balonInt: balonInt,
        concInt: document.getElementById('concInt').value,
      };

      const tipoInt = document.querySelector('input[name="tipo_Int"]:checked').value;
      datos.instIntFormat = instIntVal + (tipoInt === 'micro' ? ' (M)' : ' (PV)');

      datos.nivelesNombres = [];
      datos.nivelesInst = [];
      datos.nivelesAlicuota = [];
      datos.nivelesConc = [];
      datos.nivelesLog = [];
      datos.nivelesMv = [];

      niveles.forEach((n, idx) => {
        datos.nivelesNombres.push(n.nombre);
        datos.nivelesAlicuota.push(idx === 0 ? "NA" : n.alicuota);
        datos.nivelesConc.push(n.conc > 0 ? n.conc.toFixed(3) : "0.000");
        datos.nivelesLog.push(n.log === "N/A" ? "NA" : n.log);
        datos.nivelesMv.push(n.mv === "" ? "NA" : n.mv);
        
        if (idx === 0) {
          datos.nivelesInst.push("NA"); 
        } else {
          const tipoN = document.querySelector(`input[name="tipo_${idx}"]:checked`).value;
          const instN = document.getElementById(`inst_${idx}`).value;
          datos.nivelesInst.push(instN + (tipoN === 'micro' ? ' (M)' : ' (PV)'));
        }
      });

      datos.eqPh = document.getElementById('eqPh').value;
      datos.eqVerificacion = document.getElementById('eqVerificacion').value;
      datos.a = document.getElementById('res_a').innerText;
      datos.b = document.getElementById('res_b').innerText;
      datos.r2 = r2Text;
      datos.conformidad = document.getElementById('badgeCurva').innerText;

      // MOSTRAR PANTALLA DE CARGA CENTRAL
      document.getElementById('overlay-submit').style.display = 'flex';
      document.getElementById('submit-spinner').style.display = 'block';
      document.getElementById('submit-icon-success').style.display = 'none';
      document.getElementById('submit-title').innerText = 'Guardando curva...';
      document.getElementById('submit-title').style.color = 'var(--azul-oscuro)';
      document.getElementById('submit-text').innerText = 'Por favor, no cierre la ventana.';

      google.script.run
        .withSuccessHandler(res => {
          // TRANSICIÓN AL MENSAJE DE ÉXITO
          document.getElementById('submit-spinner').style.display = 'none';
          document.getElementById('submit-icon-success').style.display = 'block';
          
          document.getElementById('submit-title').innerText = '¡Proceso Completado!';
          document.getElementById('submit-title').style.color = 'var(--success)';
          document.getElementById('submit-text').innerText = res + '\nCerrando ventana...';
          
          setTimeout(() => google.script.host.close(), 2500);
        })
        .withFailureHandler(err => {
          // OCULTAR PANTALLA Y MOSTRAR TOAST DE ERROR
          document.getElementById('overlay-submit').style.display = 'none';
          mostrarToastError(err.message);
        })
        .guardarCurva(datos);
    }
