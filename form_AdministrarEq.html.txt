  <script>
    let DATOS = { equipos: [], areas: [] };
    let equipoSeleccionado = null;
    let modoActual = "";

    window.onload = () => {
      recargarDatos();
    };

    function recargarDatos() {
      document.getElementById('loader').style.display = 'flex';
      equipoSeleccionado = null;
      actualizarBotonesAccion();

      google.script.run
        .withSuccessHandler(res => {
          DATOS = res;
          
          const sel = document.getElementById('formArea');
          sel.innerHTML = '<option value="">Seleccione el área...</option>';
          DATOS.areas.forEach(a => sel.innerHTML += `<option value="${a}">${a}</option>`);

          const selFiltro = document.getElementById('filtroArea');
          selFiltro.innerHTML = '<option value="">-- Todas las áreas --</option>';
          DATOS.areas.forEach(a => selFiltro.innerHTML += `<option value="${a}">${a}</option>`);

          renderTabla();
          document.getElementById('loader').style.display = 'none';
        })
        .withFailureHandler(err => {
          document.getElementById('loader').style.display = 'none';
          mostrarToast(err.message, "error");
        })
        .obtenerDatosAdministracion();
    }

    function mostrarToast(msg, tipo = "success") {
      const t = document.getElementById('toast-container');
      const d = document.createElement('div');
      d.className = `toast ${tipo === 'error' ? 'toast-error' : 'toast-success'}`;
      d.innerHTML = tipo === 'error' ? `⚠️ ${msg}` : `✅ ${msg}`;
      t.appendChild(d);
      setTimeout(() => d.classList.add('show'), 10);
      setTimeout(() => { d.classList.remove('show'); setTimeout(() => d.remove(), 300); }, 3500);
    }

    // ====== MANEJO DE TABLA Y SELECCIÓN ======

    function toggleClearBtn() {
      const val = document.getElementById('buscadorTxt').value;
      document.getElementById('clearBtn').style.display = val ? 'block' : 'none';
    }

    function limpiarFiltro() {
      const input = document.getElementById('buscadorTxt');
      input.value = '';
      input.focus();
      toggleClearBtn();
      filtrarTabla();
    }

    function renderTabla() {
      const areaFiltro = document.getElementById('filtroArea').value;
      const textoFiltro = document.getElementById('buscadorTxt').value.toUpperCase().trim();
      const body = document.getElementById('bTable');
      const cont = document.getElementById('contador');
      
      body.innerHTML = "";

      const filtrados = DATOS.equipos.filter(eq => {
        let pasaArea = !areaFiltro || eq.area === areaFiltro;
        let pasaTexto = !textoFiltro || eq.codigo.toUpperCase().includes(textoFiltro);
        return pasaArea && pasaTexto;
      });
      
      if (filtrados.length === 0) {
        body.innerHTML = `<tr><td colspan="2" style="text-align:center; padding:30px; color:var(--gris); font-style:italic;">No se encontraron equipos que coincidan con la búsqueda.</td></tr>`;
        cont.innerText = `Total: 0`;
        return;
      }

      filtrados.forEach((eq, idx) => {
        let isSel = (equipoSeleccionado && equipoSeleccionado.codigo === eq.codigo) ? 'tr-selected' : '';
        body.innerHTML += `
          <tr class="${isSel}" onclick="seleccionarFila('${eq.codigo}', '${eq.area}')">
            <td><strong style="color:var(--azul-oscuro); font-size:12px;">${eq.codigo}</strong></td>
            <td>${eq.area || '<span style="color:var(--gris); font-style:italic;">No especificada</span>'}</td>
          </tr>
        `;
      });
      
      cont.innerText = `Total: ${filtrados.length}`;
    }

    function filtrarTabla() {
      equipoSeleccionado = null; 
      actualizarBotonesAccion();
      renderTabla();
    }

    function seleccionarFila(codigo, area) {
      equipoSeleccionado = { codigo: codigo, area: area };
      actualizarBotonesAccion();
      renderTabla(); 
    }

    function actualizarBotonesAccion() {
      document.getElementById('btnEditar').disabled = !equipoSeleccionado;
      document.getElementById('btnBorrar').disabled = !equipoSeleccionado;
    }

    // ====== LÓGICA DE FORMULARIOS Y MODALES ======

    function formatearCodigo(input) {
      input.value = input.value.replace(/[^A-Z0-9\-]/ig, '').toUpperCase();
    }

    function abrirModalForm(modo) {
      modoActual = modo;
      const t = document.getElementById('modalFormTitle');
      const c = document.getElementById('formCodigo');
      const a = document.getElementById('formArea');
      
      if (modo === 'agregar') {
        t.innerText = "Registrar Nuevo Equipo";
        c.value = ""; c.readOnly = false; c.classList.remove('read-only');
        a.value = "";
      } else {
        t.innerText = "Modificar Área del Equipo";
        c.value = equipoSeleccionado.codigo; c.readOnly = true; c.classList.add('read-only');
        a.value = equipoSeleccionado.area;
      }
      
      document.getElementById('modal-formulario').style.display = 'flex';
    }

    function abrirModalEliminar() {
      if(!equipoSeleccionado) return;
      document.getElementById('lblBorrarCodigo').innerText = equipoSeleccionado.codigo;
      document.getElementById('modal-eliminar').style.display = 'flex';
    }

    // ====== EJECUCIÓN CON WIZARD ======

    function iniciarCarga(titulo) {
      document.getElementById('modal-formulario').style.display = 'none';
      document.getElementById('modal-eliminar').style.display = 'none';
      
      document.getElementById('overlay-submit').style.display = 'flex';
      document.getElementById('submit-spinner').style.display = 'block';
      document.getElementById('submit-icon-success').style.display = 'none';
      document.getElementById('submit-title').innerText = titulo;
      document.getElementById('submit-title').style.color = 'var(--azul-oscuro)';
      document.getElementById('submit-text').innerText = 'Por favor, no cierre la ventana.';
    }

    function terminarCargaExito(msg) {
      document.getElementById('submit-spinner').style.display = 'none';
      document.getElementById('submit-icon-success').style.display = 'block';
      document.getElementById('submit-title').innerText = '¡Proceso Completado!';
      document.getElementById('submit-title').style.color = 'var(--success)';
      document.getElementById('submit-text').innerText = msg;
      
      setTimeout(() => {
        document.getElementById('overlay-submit').style.display = 'none';
        recargarDatos(); 
      }, 2000);
    }

    function terminarCargaError(msg) {
      document.getElementById('overlay-submit').style.display = 'none';
      mostrarToast(msg, "error");
    }

    function ejecutarGuardado() {
      const c = document.getElementById('formCodigo').value;
      const a = document.getElementById('formArea').value;

      if (!c || !a) { mostrarToast("Complete código y área.", "error"); return; }

      iniciarCarga(modoActual === 'agregar' ? "Añadiendo equipo..." : "Actualizando equipo...");

      google.script.run
        .withSuccessHandler(res => terminarCargaExito(res))
        .withFailureHandler(err => terminarCargaError(err.message))
        .gestionarEquipo({ accion: modoActual, codigo: c, area: a });
    }

    function ejecutarEliminacion() {
      iniciarCarga("Dando de baja el equipo...");
      
      google.script.run
        .withSuccessHandler(res => terminarCargaExito(res))
        .withFailureHandler(err => terminarCargaError(err.message))
        .gestionarEquipo({ accion: 'eliminar', codigo: equipoSeleccionado.codigo, area: "" });
    }

  </script>
