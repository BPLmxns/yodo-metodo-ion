  <script>
    const equipoFiltrado = "<?= equipoTarget ?>";

    function formatNum(val) {
      if (val === "" || val === null || val === "NA") return val;
      return val.toString().replace('.', ',');
    }

    function mostrarToast(msg, tipo = "success") {
      const toast = document.getElementById('toast');
      toast.innerText = tipo === "error" ? `⚠️ ${msg}` : `✅ ${msg}`;
      toast.className = tipo === "error" ? "toast-error" : "toast-success";
      toast.style.display = 'block';
      setTimeout(() => { toast.style.display = 'none'; }, 3500);
    }

    function toggleDetalle(id, btn) {
      const div = document.getElementById(id);
      const isVisible = div.style.display === 'block';
      div.style.display = isVisible ? 'none' : 'block';
      btn.querySelector('span').innerText = isVisible ? '▼' : '▲';
    }

    // ====== LLENAR SEGMENTO 1: VERIFICACIONES ======
    function llenarVerificaciones(datos) {
      document.getElementById('spin1').style.display = 'none';
      document.getElementById('cont1').style.display = 'block';
      document.getElementById('conteoConforme').innerText = datos.conforme.length;
      document.getElementById('conteoNoConforme').innerText = datos.noConforme.length;

      const tbC = document.getElementById('tbodyConforme');
      if (datos.conforme.length === 0) tbC.innerHTML = "<tr><td colspan='4'>No hay datos pendientes</td></tr>";
      else {
        tbC.innerHTML = "";
        datos.conforme.forEach(r => { 
          tbC.innerHTML += `<tr>
            <td>${r[0]}</td>
            <td><strong>${formatNum(r[1])} mV</strong></td>
            <td style="color:var(--success); font-weight:bold;">${r[2]}</td>
            <td>${r[3]}</td>
          </tr>`; 
        });
      }

      const tbNC = document.getElementById('tbodyNoConforme');
      if (datos.noConforme.length === 0) tbNC.innerHTML = "<tr><td colspan='4'>No hay datos pendientes</td></tr>";
      else {
        tbNC.innerHTML = "";
        datos.noConforme.forEach(r => { 
          tbNC.innerHTML += `<tr>
            <td>${r[0]}</td>
            <td><strong style="color:var(--danger)">${formatNum(r[1])} mV</strong></td>
            <td style="color:var(--danger); font-weight:bold;">${r[2]}</td>
            <td>${r[3]}</td>
          </tr>`; 
        });
      }
    }

    // ====== LLENAR SEGMENTO 2: REGISTRO DE USO ======
    function llenarUsos(datos) {
      document.getElementById('spin2').style.display = 'none';
      document.getElementById('cont2').style.display = 'block';
      document.getElementById('conteoUso').innerText = datos.length;

      const tb = document.getElementById('tbodyUso');
      if (datos.length === 0) tb.innerHTML = "<tr><td colspan='6'>No hay datos pendientes</td></tr>";
      else {
        tb.innerHTML = "";
        
        // Helper para pintar badges dinámicos
        const getBadge = (txt) => {
          let uTxt = String(txt).toUpperCase();
          if(uTxt.includes('NO CUMPLE')) return `<span class="badge badge-noconforme">${txt}</span>`;
          if(uTxt.includes('CUMPLE')) return `<span class="badge badge-conforme">${txt}</span>`;
          return txt; // Por defecto
        };

        datos.forEach(r => {
          tb.innerHTML += `<tr>
            <td>${r[0]}</td>
            <td><strong>${r[1]}</strong></td>
            <td style="text-align:left; color:var(--gris);">${r[2]}</td>
            <td>${getBadge(r[3])}</td>
            <td>${getBadge(r[4])}</td>
            <td>${getBadge(r[5])}</td>
          </tr>`;
        });
      }
    }

    // ====== LLENAR SEGMENTO 3: SEGUIMIENTOS ======
    function llenarSeguimientos(datos) {
      document.getElementById('spin3').style.display = 'none';
      document.getElementById('cont3').style.display = 'block';
      document.getElementById('conteoSeguimiento').innerText = datos.length;

      const tb = document.getElementById('tbodySeguimiento');
      if (datos.length === 0) tb.innerHTML = "<tr><td colspan='5'>No hay datos pendientes</td></tr>";
      else {
        tb.innerHTML = "";
        datos.forEach(r => {
          let estadoColor = (r[2] === 'Conforme') ? 'color:var(--success)' : 'color:var(--danger)';
          let aspectosTxt = r[3].replace(/; \n/g, "<br>").replace(/\n/g, "<br>");
          
          tb.innerHTML += `<tr>
            <td><strong>${r[0]}</strong></td>
            <td>${r[1]}</td>
            <td style="${estadoColor}; font-weight:bold;">${r[2]}</td>
            <td style="text-align:left; font-size:11px; line-height:1.4;">${aspectosTxt}</td>
            <td style="text-align:left;">${r[4] || '-'}</td>
          </tr>`;
        });
      }
    }

    // ====== ACCIONES ======
    function cerrarVista() {
      if (google.script.host) google.script.host.close();
      else alert("Ya no hay acciones pendientes. Puede cerrar esta pestaña de su navegador de forma segura.");
    }

    function ejecutarFirma() {
      const btn = document.getElementById('btnFirmar');
      const spin = document.getElementById('spinFirmar');
      btn.disabled = true;
      document.getElementById('txtBtn').innerText = "FIRMANDO...";
      spin.style.display = 'inline-block';

      google.script.run
        .withSuccessHandler((res) => {
          document.body.innerHTML = `
            <div style="display:flex; flex-direction:column; align-items:center; justify-content:center; height:90vh; background:white; border-radius:10px; border-top: 5px solid var(--lima);">
              <div style="font-size:50px; color:var(--azul-oscuro); margin-bottom:10px;">🛡️</div>
              <h2 style="color:var(--azul-oscuro); margin:0;">¡Firma Validada!</h2>
              <p style="color:var(--gris); font-size:13px; text-align:center; padding:0 20px; margin-bottom: 25px;">${res}</p>
              <div style="background: #f8fafc; border: 1px solid #e1e5eb; padding: 20px 30px; border-radius: 8px; text-align: center;">
                <h3 style="color: var(--success); margin: 0 0 10px 0;">Proceso Terminado</h3>
                <p style="color: var(--azul-oscuro); font-weight: 600; font-size: 14px; margin: 0;">Puede cerrar esta pestaña de su navegador con seguridad.</p>
              </div>
            </div>
          `;
          if (google.script.host) setTimeout(() => google.script.host.close(), 3000);
        })
        .withFailureHandler(err => {
          mostrarToast(err.message, "error");
          btn.disabled = false;
          document.getElementById('txtBtn').innerText = "FIRMAR Y VALIDAR";
          spin.style.display = 'none';
        })
        .firmarRegistros(equipoFiltrado);
    }

    window.onload = () => {
      if (equipoFiltrado) {
        document.getElementById('equipoLabel').innerText = "Filtrando exclusivamente el equipo: " + equipoFiltrado;
        document.getElementById('equipoLabel').style.color = "var(--azul-oscuro)";
      }
      google.script.run.withSuccessHandler(llenarVerificaciones).obtenerVerificacionesPendientes(equipoFiltrado);
      google.script.run.withSuccessHandler(llenarUsos).obtenerUsosPendientes(equipoFiltrado);
      google.script.run.withSuccessHandler(llenarSeguimientos).obtenerSeguimientoPendientes(equipoFiltrado);
    };
  </script>
