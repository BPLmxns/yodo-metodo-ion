  <script>
    let dataMatriz = {};
    
    // Ítems especializados para un pH-metro con electrodo Ion Selectivo
    const aspectos = [
      'Carcasa, panel frontal y display limpios y sin daños',
      'Membrana sensora limpia, sin rayaduras profundas ni desgaste excesivo',
      'Teclado / botones operando suavemente y sin trabas',
      'Cables de alimentación y de electrodos sin cortes ni exposición',
      'Conectores BNC/Din limpio, libre de humedad, polvo u oxidación',
      'Brazo articulado / soporte de electrodos firme y estable',
      'Cuerpo del electrodo ISE sin grietas, desgaste mecánico ni fugas',
      'Solución interna / electrolito libre de cristalización o contaminación',
      'Lectura de señal (mV) estable, sin fluctuaciones erraticas al sumergir',
      'Etiqueta de calibración / mantenimiento vigente y legible'
    ];

    window.onload = () => {
      const hoy = new Date();
      document.getElementById('fecha').value = hoy.getFullYear() + '-' + String(hoy.getMonth() + 1).padStart(2, '0') + '-' + String(hoy.getDate()).padStart(2, '0');

      const tbody = document.querySelector('#tablaChecklist tbody');
      aspectos.forEach((txt, i) => {
        tbody.insertAdjacentHTML('beforeend', `
          <tr id="row${i}">
            <td class="aspecto">${txt}</td>
            <td><input type="radio" name="chk${i}" value="Conforme" onchange="calcularConformidad(${i})"></td>
            <td><input type="radio" name="chk${i}" value="No conforme" onchange="calcularConformidad(${i})"></td>
          </tr>
        `);
      });

      google.script.run
        .withSuccessHandler(datos => {
          dataMatriz = datos;
          poblarEquipos();
          document.getElementById('loader').style.display = 'none';
          asignarEventosValidacion();
        })
        .obtenerDatosMatriz();
    };

    function poblarEquipos() {
      const lista = document.getElementById('listaOpciones');
      lista.innerHTML = '';
      dataMatriz.equipos.forEach(eq => {
        let div = document.createElement('div');
        div.className = 'dropdown-option';
        div.innerText = eq.codigo;
        div.onclick = () => {
          document.getElementById('equipoCodigo').value = eq.codigo;
          document.getElementById('equipoCodigo').classList.remove('invalid');
          lista.classList.remove('show');
          seleccionarEquipo();
        };
        lista.appendChild(div);
      });
    }

    function mostrarOpciones() {
      document.getElementById('listaOpciones').classList.add('show');
      filtrarOpciones();
    }

    function filtrarOpciones() {
      const input = document.getElementById('equipoCodigo').value.toUpperCase();
      const lista = document.getElementById('listaOpciones');
      const opciones = lista.getElementsByClassName('dropdown-option');
      let visibles = 0;

      for (let i = 0; i < opciones.length; i++) {
        let txtValue = opciones[i].innerText;
        if (txtValue.toUpperCase().indexOf(input) > -1) {
          opciones[i].style.display = "";
          visibles++;
        } else {
          opciones[i].style.display = "none";
        }
      }

      let noRes = document.getElementById('no-results-msg');
      if (visibles === 0) {
        if (!noRes) {
          noRes = document.createElement('div');
          noRes.id = 'no-results-msg';
          noRes.className = 'no-results';
          noRes.innerText = 'No se encontró el equipo';
          lista.appendChild(noRes);
        }
      } else {
        if (noRes) noRes.remove();
      }
      
      seleccionarEquipo();
    }

    document.addEventListener('click', function(event) {
      const container = document.querySelector('.dropdown-container');
      if (!container.contains(event.target)) {
        document.getElementById('listaOpciones').classList.remove('show');
      }
    });

    function seleccionarEquipo() {
      const cod = document.getElementById('equipoCodigo').value;
      const eq = dataMatriz.equipos.find(x => x.codigo === cod);
      if(eq) {
        document.getElementById('equipoArea').value = eq.area;
      } else {
        document.getElementById('equipoArea').value = '';
      }
    }

    function mostrarToastError(mensaje) {
      const container = document.getElementById('toast-container');
      const toast = document.createElement('div');
      toast.className = `toast toast-error`;
      toast.innerHTML = `⚠️ ${mensaje}`;
      container.appendChild(toast);
      setTimeout(() => toast.classList.add('show'), 10);
      setTimeout(() => { toast.classList.remove('show'); setTimeout(() => toast.remove(), 300); }, 3500);
    }

    function asignarEventosValidacion() {
      const inputs = document.querySelectorAll('input:not([type="radio"]), textarea');
      inputs.forEach(input => {
        input.addEventListener('input', function() {
          if (this.value.trim() !== "") this.classList.remove('invalid');
        });
      });
    }

    function calcularConformidad(rowIndex) {
      document.getElementById(`row${rowIndex}`).classList.remove('row-invalid');

      let evaluados = 0;
      let noConforme = false;
      
      aspectos.forEach((_, i) => {
        const seleccionado =[...document.getElementsByName(`chk${i}`)].find(r => r.checked);
        if (seleccionado) {
          evaluados++;
          if (seleccionado.value === 'No conforme') noConforme = true;
        }
      });

      const badge = document.getElementById('statusBadge');
      if (evaluados === aspectos.length) {
        badge.innerText = noConforme ? 'NO CONFORME' : 'CONFORME';
        badge.className = `badge ${noConforme ? 'badge-noconforme' : 'badge-conforme'}`;
      } else {
        badge.innerText = `Evaluados ${evaluados}/${aspectos.length}`;
        badge.className = 'badge';
      }
    }

    function intentarRegistrar() {
      const cod = document.getElementById('equipoCodigo');
      const eq = dataMatriz.equipos.find(x => x.codigo === cod.value);
      
      let hayErrores = false;

      // 1. Validar Código
      if (!cod.value || !eq) {
        cod.classList.add('invalid');
        mostrarToastError("Seleccione un equipo válido de la lista");
        hayErrores = true;
      }

      // 2. Validar Checklist
      let faltanCheck = false;
      aspectos.forEach((_, i) => {
        const checkeado = [...document.getElementsByName(`chk${i}`)].some(r => r.checked);
        if (!checkeado) {
          document.getElementById(`row${i}`).classList.add('row-invalid');
          faltanCheck = true;
        }
      });

      if (faltanCheck) {
        mostrarToastError("Complete todo el checklist de condiciones");
        hayErrores = true;
      }

      // 3. Validar Textareas
      const obs = document.getElementById('observaciones');
      const corr = document.getElementById('correcciones');

      if (obs.value.trim() === '') {
        obs.classList.add('invalid');
        hayErrores = true;
      }
      if (corr.value.trim() === '') {
        corr.classList.add('invalid');
        hayErrores = true;
      }

      if(hayErrores && (obs.value.trim() === '' || corr.value.trim() === '')){
         mostrarToastError("Todos los campos de texto son obligatorios");
      }

      if (hayErrores) return;

      const estadoFinal = document.getElementById('statusBadge').innerText;
      const detalleAspectos = aspectos.map((txt, i) => {
        const rad =[...document.getElementsByName(`chk${i}`)].find(x => x.checked);
        return `${txt} (${rad.value})`;
      }).join('; \n');

      const datos = {
        codigo: cod.value,
        area: document.getElementById('equipoArea').value,
        estado: estadoFinal === 'CONFORME' ? 'Conforme' : 'No conforme',
        aspectos: detalleAspectos,
        observaciones: obs.value.trim(),
        correcciones: corr.value.trim()
      };

      // MOSTRAR PANTALLA DE CARGA CENTRAL
      document.getElementById('overlay-submit').style.display = 'flex';
      document.getElementById('submit-spinner').style.display = 'block';
      document.getElementById('submit-icon-success').style.display = 'none';
      document.getElementById('submit-title').innerText = 'Registrando información...';
      document.getElementById('submit-title').style.color = 'var(--azul-oscuro)';
      document.getElementById('submit-text').innerText = 'Por favor, no cierre la ventana.';

      google.script.run
        .withSuccessHandler((res) => {
          // TRANSICIÓN AL MENSAJE DE ÉXITO
          document.getElementById('submit-spinner').style.display = 'none';
          document.getElementById('submit-icon-success').style.display = 'block';
          
          document.getElementById('submit-title').innerText = '¡Proceso Completado!';
          document.getElementById('submit-title').style.color = 'var(--success)';
          
          document.getElementById('submit-text').innerText = res + '\nCerrando ventana...';
          
          // Cerrar automáticamente después de mostrar el éxito
          setTimeout(() => google.script.host.close(), 2500);
        })
        .withFailureHandler((err) => {
          // OCULTAR PANTALLA Y MOSTRAR TOAST DE ERROR
          document.getElementById('overlay-submit').style.display = 'none';
          mostrarToastError(err.message);
        })
        .registrarSeguimiento(datos);
    }
  </script>
