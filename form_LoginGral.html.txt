  <script>
    window.onload = () => {
      // 1. Cargar el correo del usuario
      google.script.run
        .withSuccessHandler(e => document.getElementById('userEmail').innerText = e)
        .obtenerUsuarioActual();

      // 2. Habilitar la tecla "Enter" para el campo de contraseña
      document.getElementById('passInput').addEventListener('keyup', function(event) {
        if (event.key === 'Enter') {
          validarYEntrar();
        }
      });
      
      // Auto-focus en la contraseña
      document.getElementById('passInput').focus();
    };

    function validarYEntrar() {
      const pass = document.getElementById('passInput').value;
      const btn = document.getElementById('btnAutorizar');
      const msg = document.getElementById('msgAuth');

      if (!pass) { 
        msg.innerText = "⚠️ Ingrese su contraseña"; 
        msg.style.color = "#ef4444"; 
        msg.style.background = "#fee2e2"; 
        msg.style.display = "block"; 
        return; 
      }

      btn.disabled = true; 
      btn.innerText = "VERIFICANDO..."; 
      msg.style.display = "none";

      // ENCENDER WIZARD DE CARGA
      document.getElementById('overlay-submit').style.display = 'flex';
      document.getElementById('submit-spinner').style.display = 'block';
      document.getElementById('submit-icon-success').style.display = 'none';
      document.getElementById('submit-title').innerText = 'Verificando credenciales...';
      document.getElementById('submit-title').style.color = 'var(--azul-oscuro)';
      document.getElementById('submit-text').innerText = 'Validando accesos en base de datos.';

      google.script.run
        .withSuccessHandler((res) => {
          // TRANSICIÓN AL ÉXITO
          document.getElementById('submit-spinner').style.display = 'none';
          document.getElementById('submit-icon-success').style.display = 'block';
          document.getElementById('submit-title').innerText = '¡Acceso Concedido!';
          document.getElementById('submit-title').style.color = 'var(--success)';
          document.getElementById('submit-text').innerText = 'Abriendo panel operativo...';
          
          setTimeout(() => {
            google.script.run
              .withSuccessHandler(() => { google.script.host.close(); })
              .redirigirYAbrirPanel();
          }, 1500);
        })
        .withFailureHandler((err) => {
          // APAGAR WIZARD Y MOSTRAR ALERTA ROJA EN LA TARJETA
          document.getElementById('overlay-submit').style.display = 'none';
          btn.disabled = false; 
          btn.innerText = "INGRESAR AL SISTEMA";
          msg.innerText = "❌ " + err.message; 
          msg.style.color = "#ef4444"; 
          msg.style.background = "#fee2e2"; 
          msg.style.display = "block";
        })
        .iniciarSesionSGC(pass);
    }
  </script>
