    let datosFinales = {};

    window.onload = () => {
      const ahora = new Date();
      document.getElementById('fechaHora').value = `${ahora.getFullYear()}-${String(ahora.getMonth() + 1).padStart(2, '0')}-${String(ahora.getDate()).padStart(2, '0')} - ${String(ahora.getHours()).padStart(2, '0')}:${String(ahora.getMinutes()).padStart(2, '0')}`;
      
      google.script.run.withSuccessHandler(res => consolidarCarga("muestras", res)).withFailureHandler(errorCarga).getMuestrasData();
      google.script.run.withSuccessHandler(res => consolidarCarga("inventario", res)).withFailureHandler(errorCarga).getInventarioData();
      google.script.run.withSuccessHandler(res => consolidarCarga("curvas", res)).withFailureHandler(errorCarga).getCurvasData();
    };

    function errorCarga(err) {
      document.getElementById('loader').style.display='none';
      mostrarToast(err.message, "error");
    }

    function consolidarCarga(tipo, data) {
      datosFinales[tipo] = data;
      procesosCargados++;
      if (procesosCargados === procesosTotales) {
        const consolidado = {
          muestras: datosFinales.muestras.muestras,
          muestrasVerificacion: datosFinales.muestras.muestrasVerificacion,
          mapReplicas: datosFinales.muestras.mapReplicas,
          controlesRutina: datosFinales.muestras.controlesRutina, 
          balanzas: datosFinales.inventario.balanzas,
          pipetas10mL: datosFinales.inventario.pipetas10mL,
          balones100mL: datosFinales.inventario.balones100mL,
          lotesISA: datosFinales.inventario.lotesISA,
          curvasHoy: datosFinales.curvas
        };
        inicializarFormulario(consolidado);
      }
    }

    function mostrarToast(msg, tipo) {
      const t = document.getElementById('toast-container');
      const d = document.createElement('div');
      d.className = `toast ${tipo === 'error' ? 'toast-error' : 'toast-success'}`;
      d.innerHTML = tipo === 'error' ? `⚠️ ${msg}` : `✅ ${msg}`;
      t.appendChild(d);
      setTimeout(() => d.classList.add('show'), 10);
      setTimeout(() => { d.classList.remove('show'); setTimeout(() => d.remove(), 300); }, 3000);
    }

      for (let i = 0; i < muestrasBatch.length; i++) {

    function guardarEstadoTabla() {
      for (let i = 1; i <= numFilas; i++) {
        const tipoEl = document.getElementById(`tipo_${i}`);
        if (!tipoEl) continue;

        const tipoVal = tipoEl.value;
        const usaCombo = (tipoVal === 'Rutina' || tipoVal === 'Verificación');
        
        const inf = document.getElementById(`inf_${i}`);
        const sup = document.getElementById(`sup_${i}`);
        const chk = document.getElementById(`na_${i}`).checked;

        if (muestrasBatch[(i-1)*2]) {
          muestrasBatch[(i-1)*2].inf = inf ? inf.value : "";
          muestrasBatch[(i-1)*2].sup = sup ? sup.value : "";
          muestrasBatch[(i-1)*2].na = chk;
          muestrasBatch[(i-1)*2 + 1].inf = inf ? inf.value : "";
          muestrasBatch[(i-1)*2 + 1].sup = sup ? sup.value : "";
          muestrasBatch[(i-1)*2 + 1].na = chk;
        }
      }
    }

    // ================= FUNCIONES DEL DROPDOWN MEJORADO =================
    function poblarComboGeneral(arr, listId, inputId, isObj=false, idxObj=null) {
      const l = document.getElementById(listId); l.innerHTML = '';
      if(arr.length===0) { l.innerHTML = '<div class="no-results">Sin datos</div>'; return; }
      arr.forEach(i => {
        let d = document.createElement('div'); d.className = 'dropdown-option'; d.innerText = i;
        d.onclick = () => { 
          document.getElementById(inputId).value = i; 
          document.getElementById(inputId).classList.remove('invalid'); 
          l.classList.remove('show'); 
          const clearBtn = document.getElementById('clear_' + inputId);
          if(clearBtn) clearBtn.style.display = 'block';
          if(isObj) seleccionarControl(idxObj); 
        };
        l.appendChild(d);
      });
    }

    function mostrarOpcionesGeneral(lId, iId) { 
      document.getElementById(lId).classList.add('show'); 
      filtrarOpcionesGeneral(lId, iId); 
    }

    function filtrarOpcionesGeneral(lId, iId) {
      const v = document.getElementById(iId).value.toUpperCase();
      const l = document.getElementById(lId);
      const ops = l.getElementsByClassName('dropdown-option');
      let vis=0;
      for(let i=0; i<ops.length; i++) { 
        if(ops[i].innerText.toUpperCase().indexOf(v)>-1) { ops[i].style.display=""; vis++; } 
        else { ops[i].style.display="none"; }
      }
      let nr = l.querySelector('.no-results');
      if(vis===0 && ops.length>0) { 
        if(!nr) { nr=document.createElement('div'); nr.className='no-results'; nr.innerText='Sin coincidencias'; l.appendChild(nr); } 
      } else if(nr && vis>0) { nr.remove(); }

      const clearBtn = document.getElementById('clear_' + iId);
      if(clearBtn) clearBtn.style.display = v ? 'block' : 'none';

      const inputEl = document.getElementById(iId);
      if(inputEl && l.classList.contains('show')) {
        const rect = inputEl.getBoundingClientRect();
        const windowHeight = window.innerHeight;
        if (rect.bottom + 160 > windowHeight && rect.top > 160) {
          l.style.top = 'auto'; l.style.bottom = '100%'; l.style.marginTop = '0'; l.style.marginBottom = '4px'; l.style.boxShadow = '0 -4px 12px rgba(0,0,0,0.15)';
        } else {
          l.style.top = '100%'; l.style.bottom = 'auto'; l.style.marginTop = '4px'; l.style.marginBottom = '0'; l.style.boxShadow = '0 4px 12px rgba(0,0,0,0.15)';
        }
      }
    }

    function limpiarDropdownGeneral(listId, inputId, isObj=false, idxObj=null) {
      const input = document.getElementById(inputId);
      input.value = '';
      input.focus();
      filtrarOpcionesGeneral(listId, inputId);
      if(isObj) seleccionarControl(idxObj); 
    }

    function ajustarTablaMuestras() {
      const val = parseInt(document.getElementById('numMuestras').value) || 0;
      const tbody = document.getElementById('tbodyMuestras');
      
      if (numFilas === 0 && val > 0) tbody.innerHTML = ''; 
      
      if (val > numFilas) {
        for (let i = numFilas + 1; i <= val; i++) {
          const tr = document.createElement('tr'); tr.id = `row_${i}`; tr.innerHTML = crearFilaHTML(i);
          tbody.appendChild(tr); 
          cambiarTipoMuestraFila(i); 
        }
      } else if (val < numFilas && val >= 0) {
        for (let i = numFilas; i > val; i--) { const tr = document.getElementById(`row_${i}`); if (tr) tr.remove(); }
      }
      
      numFilas = val;
      if (numFilas === 0) {
        tbody.innerHTML = '<tr><td colspan="5" style="text-align:center; padding:20px; color:var(--gris); font-style:italic;">Digite el número de muestras.</td></tr>';
      }
    }

    function cambiarTipoMuestraFila(idx) {
      const tipoSel = document.getElementById(`tipo_${idx}`);
      const tipoVal = tipoSel.value;
      
      if(tipoVal === 'Rutina') { tipoSel.style.color = 'var(--azul-oscuro)'; tipoSel.style.backgroundColor = '#f0f9ff'; tipoSel.style.borderColor = '#bae6fd'; }
      else if(tipoVal === 'Verificación') { tipoSel.style.color = '#059669'; tipoSel.style.backgroundColor = '#ecfdf5'; tipoSel.style.borderColor = '#a7f3d0'; }
      else if(tipoVal === 'Validación') { tipoSel.style.color = '#d97706'; tipoSel.style.backgroundColor = '#fffbeb'; tipoSel.style.borderColor = '#fde68a'; }
      else { tipoSel.style.color = '#6b7280'; tipoSel.style.backgroundColor = '#f3f4f6'; tipoSel.style.borderColor = '#e5e7eb'; }

      const usaCombo = (tipoVal === 'Rutina' || tipoVal === 'Verificación');
      const comboDiv = document.getElementById(`box_combo_${idx}`);
      const inputLibre = document.getElementById(`txt_libre_${idx}`);
      const prodCell = document.getElementById(`prod_${idx}`);
      const comboEl = document.getElementById(`combo_${idx}`);
      const clearBtn = document.getElementById(`clear_combo_${idx}`);
      const guiaArea = document.getElementById(`guia_${idx}`);
      
      comboEl.value = ""; 
      inputLibre.value = "";
      prodCell.value = "";
      if(clearBtn) clearBtn.style.display = 'none';
      if(guiaArea) { guiaArea.innerText = "Esperando..."; guiaArea.style.color = "var(--gris)"; }
      
      document.getElementById(`na_${idx}`).checked = false; toggleNA(idx);
      document.getElementById(`inf_${idx}`).value = ""; document.getElementById(`sup_${idx}`).value = "";

      if (usaCombo) {
        comboDiv.style.display = "block"; inputLibre.style.display = "none";
        prodCell.classList.add('read-only'); prodCell.readOnly = true;
        const dbActiva = (tipoVal === 'Verificación') ? dbVerificacion : dbMuestras;
        poblarComboGeneral(dbActiva.map(m=>m.control), `lista_${idx}`, `combo_${idx}`, true, idx);
      } else {
        comboDiv.style.display = "none"; inputLibre.style.display = "block";
        prodCell.classList.remove('read-only'); prodCell.readOnly = false;
        if(guiaArea) guiaArea.innerText = "Edición libre habilitada";
      }
    }

    function toggleNA(idx) {
      const isC = document.getElementById(`na_${idx}`).checked, inf = document.getElementById(`inf_${idx}`), sup = document.getElementById(`sup_${idx}`);
      inf.disabled = isC; sup.disabled = isC;
      if(isC) { 
        inf.classList.add('read-only'); sup.classList.add('read-only'); 
        inf.classList.remove('invalid'); sup.classList.remove('invalid'); 
        inf.value = ""; sup.value = "";
      } else { 
        inf.classList.remove('read-only'); sup.classList.remove('read-only'); 
      }
    }

    function seleccionarControl(idx) {
      const cCod = document.getElementById(`combo_${idx}`).value;
      const pArea = document.getElementById(`prod_${idx}`);
      const gArea = document.getElementById(`guia_${idx}`);
      const inf = document.getElementById(`inf_${idx}`);
      const sup = document.getElementById(`sup_${idx}`);
      const naChk = document.getElementById(`na_${idx}`);
      
      const tipoVal = document.getElementById(`tipo_${idx}`).value;
      
      if (tipoVal === 'Rutina' && controlesRutina.includes(cCod)) {
        document.getElementById(`combo_${idx}`).value = "";
        const clearBtn = document.getElementById(`clear_combo_${idx}`);
        if(clearBtn) clearBtn.style.display = 'none';
        pArea.value = ""; gArea.innerText = "Esperando..."; gArea.style.color = "var(--gris)"; 
        mostrarModalAlertaRutina(cCod);
        return;
      }

      const dbActiva = (tipoVal === 'Verificación') ? dbVerificacion : dbMuestras;
      const data = dbActiva.find(m => m.control === cCod);
      
      if (data) { 
        pArea.value = data.producto; 
        gArea.innerText = (tipoVal === 'Verificación') ? "Verificación de muestra anterior" : ("LIMS: " + data.especificacion); 
        gArea.style.color = "var(--azul-oscuro)"; 
        pArea.classList.remove('invalid');

        if (tipoVal === 'Verificación') {
          if (data.inf && data.inf.toString().toUpperCase().includes("NO APLICA")) { naChk.checked = true; } 
          else { naChk.checked = false; inf.value = data.inf ? data.inf.toString().replace(',','.') : ""; sup.value = data.sup ? data.sup.toString().replace(',','.') : ""; }
          toggleNA(idx);
        } else {
          inf.value = ""; sup.value = ""; naChk.checked = false; toggleNA(idx);
        }
      } else { 
        pArea.value = ""; gArea.innerText = "No encontrado"; gArea.style.color = "var(--error)"; 
      }
    }

    // ================= TRANSICIÓN S1 -> S2 =================
    function irSiguiente1() {
      if (numFilas === 0) { mostrarToast("Debe agregar muestras.", "error"); return; }
      
      let err = false;
      let mapTempRep = Object.assign({}, mapReplicas); 
      let newMuestras = [];
      let counts = { "Rutina": 0, "Verificación": 0, "Validación": 0, "Otros": 0 };

      // 🛠️ BUGFIX: Validación de vacíos corregida y blindada
      for (let i = 1; i <= numFilas; i++) {
        const tipoVal = document.getElementById(`tipo_${i}`).value;
        const usaCombo = (tipoVal === 'Rutina' || tipoVal === 'Verificación');
        
        const ctrl = usaCombo ? document.getElementById(`combo_${i}`) : document.getElementById(`txt_libre_${i}`);
        const prod = document.getElementById(`prod_${i}`);
        const inf = document.getElementById(`inf_${i}`);
        const sup = document.getElementById(`sup_${i}`);
        const chk = document.getElementById(`na_${i}`).checked;

        if(!ctrl.value || !ctrl.value.trim()) { ctrl.classList.add('invalid'); err=true; } else ctrl.classList.remove('invalid');
        if(!prod.value || !prod.value.trim()) { prod.classList.add('invalid'); err=true; } else prod.classList.remove('invalid');
        if(!chk) {
          if(!inf.value) { inf.classList.add('invalid'); err=true; } else inf.classList.remove('invalid');
          if(!sup.value) { sup.classList.add('invalid'); err=true; } else sup.classList.remove('invalid');
        }

        let cTxt = ctrl.value ? ctrl.value.trim() : ""; 
        let pTxt = prod.value ? prod.value.trim() : "";
        let repBase = mapTempRep[cTxt] || 0;
        let r1 = repBase + 1, r2 = repBase + 2;
        mapTempRep[cTxt] = r2; 
        counts[tipoVal]++;

        const dbActiva = (tipoVal === 'Verificación') ? dbVerificacion : dbMuestras;
        let idenTxt = "N/A";
        const baseM = dbActiva.find(m => m.control === cTxt);
        if(baseM && baseM.identificacion) idenTxt = baseM.identificacion;

        let old1 = muestrasBatch[(i-1)*2] || {}; let old2 = muestrasBatch[(i-1)*2 + 1] || {};

        newMuestras.push({ 
          tipo: tipoVal, c: cTxt, p: pTxt, r: r1, iden: idenTxt, inf: inf.value, sup: sup.value, na: chk,
          pso: old1.pso || "", bal: old1.bal || "", bM: old1.bM || "", pip: old1.pip || "", bD: old1.bD || "", isa: old1.isa || "", mv: old1.mv || "", mgkg: old1.mgkg || "" 
        });
        newMuestras.push({ 
          tipo: tipoVal, c: cTxt, p: pTxt, r: r2, iden: idenTxt, inf: inf.value, sup: sup.value, na: chk,
          pso: old2.pso || "", bal: old2.bal || "", bM: old2.bM || "", pip: old2.pip || "", bD: old2.bD || "", isa: old2.isa || "", mv: old2.mv || "", mgkg: old2.mgkg || "" 
        });
      }

      if (err) { mostrarToast("Complete todos los campos del Segmento 1 para avanzar.", "error"); return; }
      muestrasBatch = newMuestras; 

      let summaryText = Object.entries(counts).filter(([k,v]) => v > 0).map(([k,v]) => `${v} ${k}`).join(', ');
      
      document.getElementById('resFecha').value = document.getElementById('fechaHora').value;
      document.getElementById('resTipo').value = summaryText;
      document.getElementById('resTotal').value = muestrasBatch.length;

      armarTablaPrep();

      document.getElementById('segmento1').classList.remove('active'); document.getElementById('segmento2').classList.add('active');
      document.getElementById('step1').classList.remove('active'); document.getElementById('step2').classList.add('active');
      
      document.getElementById('btnCerrar').style.display = 'none'; document.getElementById('btnSiguienteS1').style.display = 'none';
      document.getElementById('btnAtras1').style.display = 'flex'; document.getElementById('btnSiguienteS2').style.display = 'flex';
    }

    function calcularConcentracion(idx) {
      const valMv = parseFloat(document.getElementById(`mv_res_${idx}`).value);
      const celdaMg = document.getElementById(`mgkg_res_${idx}`);
      const pairIndex = Math.floor(idx / 2);
      
      if (isNaN(valMv) || !curvaActiva) { 
        celdaMg.innerText = "-"; 
        calcularPar(pairIndex);
        return; 
      }

      const mvBlanco = parseFloat(curvaActiva.mv[0]) || 0;
      const yCorr = valMv - mvBlanco;
      const logC = (yCorr - curvaActiva.b) / curvaActiva.a;
      const cRead = Math.pow(10, logC);

      const w = parseFloat(muestrasBatch[idx].pso);
      const factorDilucion = 1000 / w; 
      const cFinal = cRead * factorDilucion;
      
      celdaMg.innerText = cFinal.toFixed(2);
      calcularPar(pairIndex);
    }

    function calcularPar(pairIndex) {
      const idx1 = pairIndex * 2; const idx2 = pairIndex * 2 + 1;
      const txt1 = document.getElementById(`mgkg_res_${idx1}`).innerText;
      const txt2 = document.getElementById(`mgkg_res_${idx2}`).innerText;
      const m1 = muestrasBatch[idx1]; 
      
      const cProm = document.getElementById(`prom_${pairIndex}`);
      const cPR = document.getElementById(`prsdr_R_${pairIndex}`);
      const cRSDr = document.getElementById(`rsd_r_${pairIndex}`); 
      const cHor = document.getElementById(`horrat_${pairIndex}`);
      const cHorBadge = document.getElementById(`horrat_badge_${pairIndex}`);
      const cConfEsp = document.getElementById(`conf_esp_${pairIndex}`);
      
      if (txt1 === "-" || txt2 === "-" || !txt1 || !txt2) {
        cProm.innerText = "-"; cPR.innerText = "-"; cRSDr.innerText = "-"; cHor.innerText = "-"; cHorBadge.innerHTML = "";
        cConfEsp.innerHTML = "<span class='badge badge-pendiente'>Pendiente</span>"; 
        return;
      }
      
      const c1 = parseFloat(txt1), c2 = parseFloat(txt2);
      const prom = (c1 + c2) / 2;
      
      cProm.innerText = prom.toFixed(2);
      
      if (m1.na) {
        cConfEsp.innerHTML = "<span class='badge' style='background:#f1f5f9; color:#64748b; border-color:#cbd5e1;'>NO APLICA</span>";
      } else {
        const inf = parseFloat(m1.inf);
        const sup = parseFloat(m1.sup);
        if (prom >= inf && prom <= sup) {
          cConfEsp.innerHTML = "<span class='badge badge-conforme'>CUMPLE</span>";
        } else {
          cConfEsp.innerHTML = "<span class='badge badge-noconforme'>NO CUMPLE</span>";
        }
      }
      
      if (prom === 0) {
         cPR.innerText = "0.00"; cRSDr.innerText = "0.00"; cHor.innerText = "0.00";
         cHorBadge.innerHTML = "<span class='badge badge-conforme' style='padding:2px 4px; border:none; margin-top:2px;'>CUMPLE</span>"; 
         return;
      }
      
      const C = 0.000001; 
      const c_calc = C * prom; 
      
      let Sr = 0;
      if (c_calc < 0.00000012) Sr = 0.22 * c_calc;
      else if (c_calc > 0.138) Sr = Math.sqrt(c_calc) * 0.01;
      else Sr = Math.pow(c_calc, 0.8495) * 0.02;

      const prsdR = (Sr / c_calc) * 100;
      const sd_exp = Math.abs(c1 - c2) / Math.SQRT2;
      const rsd_r = (sd_exp / prom) * 100;
      
      const horrat = rsd_r / prsdR;
      
      cPR.innerText = prsdR.toFixed(2);
      cRSDr.innerText = rsd_r.toFixed(2); 
      cHor.innerText = horrat.toFixed(2);
      
      if (horrat <= 1.3) {
         cHorBadge.innerHTML = "<span class='badge badge-conforme' style='padding:2px 4px; border:none; margin-top:2px;'>CUMPLE</span>";
      } else {
         cHorBadge.innerHTML = "<span class='badge badge-noconforme' style='padding:2px 4px; border:none; margin-top:2px;'>NO CUMPLE</span>";
      }
    }
