// --- 🚀 PUENTE PARA ESCRITURA ---

function registrarSeguimiento(datos) {
  const usuario = Session.getActiveUser().getEmail();
  const payload = { accion: "registrarSeguimiento", datos: datos, usuario: usuario };
  
  // 1. Enviamos el POST inicial evadiendo la redirección automática
  let opcionesPost = {
    method: "post",
    contentType: "application/json",
    headers: { "Authorization": "Bearer " + ScriptApp.getOAuthToken() },
    payload: JSON.stringify(payload),
    muteHttpExceptions: true,
    followRedirects: false 
  };
  
  let resp = UrlFetchApp.fetch(CONFIG.WEB_APP_URL, opcionesPost);
  
  // 2. Si Google responde con un salto (302), atrapamos la URL temporal
  if (resp.getResponseCode() === 302 || resp.getResponseCode() === 303 || resp.getResponseCode() === 307) {
    const nuevaUrl = resp.getHeaders()['Location'];
    
    // 3. ✅ EL SECRETO: Hacemos el salto mediante un 'GET' limpio, inyectando el token nuevamente
    let opcionesGet = {
      method: "get",
      headers: { "Authorization": "Bearer " + ScriptApp.getOAuthToken() },
      muteHttpExceptions: true
    };
    resp = UrlFetchApp.fetch(nuevaUrl, opcionesGet);
  }
  
  const textoRespuesta = resp.getContentText();
  
  if (textoRespuesta.trim().toLowerCase().startsWith("<!doctype")) {
    throw new Error("Error del servidor de Google al retornar los datos. La operación posiblemente se guardó.");
  }
  
  const resultado = JSON.parse(textoRespuesta);
  if (!resultado.exito) throw new Error(resultado.error);
  
  return resultado.datos;
}

// --- 🔐 LÓGICA REAL DE ESCRITURA (En la Web App) ---

function ejecutarLogicaRegistrarSeguimiento(datos, usuario) {
  try {
    const ss = SpreadsheetApp.openById(CONFIG.EQUIPO_PH.ID);
    const hoja = ss.getSheetByName(CONFIG.EQUIPO_PH.HOJAS.SEGUIMIENTO);
    if (!hoja) throw new Error("No se encontró la hoja 'Seguimiento'.");
    
    const fechaObj = new Date();
    const fechaActual = Utilities.formatDate(fechaObj, Session.getScriptTimeZone(), "yyyy-MM-dd");
    const aspectosFormateados = datos.aspectos + ".";
    
    hoja.appendRow([ fechaActual, datos.estado, aspectosFormateados, datos.observaciones, datos.correcciones, usuario, "", "" ]);
    
    const ultimaFila = hoja.getLastRow();
    const celdaEstado = hoja.getRange(ultimaFila, 2);
    if (datos.estado === 'Conforme') celdaEstado.setBackground('#ecfdf5').setFontColor('#059669').setFontWeight('bold');
    else celdaEstado.setBackground('#fef2f2').setFontColor('#dc2626').setFontWeight('bold');
    hoja.getRange(ultimaFila, 3).setWrap(true);

    const urlDocumento = ss.getUrl();
    enviarCorreoSeguimiento(datos.codigo, datos.area, fechaObj, datos.estado, datos.observaciones, urlDocumento);
    
    return "✅ Seguimiento registrado exitosamente. Correo enviado al Liderazgo.";
  } catch (error) { throw new Error(error.message); }
}
