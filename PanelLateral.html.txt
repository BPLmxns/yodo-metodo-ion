      function ejecutar(btn, nombreFuncion) {
        const spinner = btn.querySelector('.spinner');
        btn.disabled = true;
        if(spinner) spinner.style.display = 'inline-block';

        google.script.run
          .withSuccessHandler(() => {
             google.script.run
                .withSuccessHandler((respuesta) => {
                  btn.disabled = false;
                  if(spinner) spinner.style.display = 'none';
                  
                  if(typeof respuesta === 'object' && respuesta !== null) {
                    if(respuesta.estado === "ALERTA") {
                      abrirModal(); 
                    } else if (respuesta.estado === "ALERTA_VERIFICACION") {
                      abrirModalVerif(respuesta.mensaje); 
                    } else if (respuesta.estado === "ALERTA_CURVA") {
                      abrirModalCurva(respuesta.mensaje); 
                    } else if (respuesta.estado === "OK") {
                      mostrarMensaje("✅ " + respuesta.mensaje, "exito");
                    }
                  } else {
                    mostrarMensaje("✅ " + respuesta, "exito");
                  }
                })
                .withFailureHandler((err) => {
                  btn.disabled = false;
                  if(spinner) spinner.style.display = 'none';
                  mostrarMensaje("❌ Error: " + err.message, "error");
                })[nombreFuncion]();
          })
          .withFailureHandler((err) => {
             btn.disabled = false;
             if(spinner) spinner.style.display = 'none';
             if (err.message === "SESION_EXPIRADA") {
                google.script.host.close();
                google.script.run.entrarAlSistema();
             } else {
                mostrarMensaje("❌ Error de sesión: " + err.message, "error");
             }
          })
          .verificarSesion(); 
      }

      function mostrarMensaje(texto, tipo) {
        const toast = document.getElementById('toast');
        toast.innerText = texto;
        toast.style.backgroundColor = (tipo === "error") ? "#ef4444" : "#0a4079";
        toast.style.display = 'block';
        
        setTimeout(() => { toast.style.display = 'none'; }, 4000);
      }

      /* ==============================================
         Funciones del Modal de Seguimiento
         ============================================== */
      function abrirModal() { document.getElementById('modal-alerta').style.display = 'flex'; }
      function cerrarModal() { document.getElementById('modal-alerta').style.display = 'none'; }

      function ejecutarSeguimientoMensual(btn) {
        const spinner = btn.querySelector('.spinner');
        btn.disabled = true;
        if(spinner) spinner.style.display = 'inline-block';

        google.script.run
          .withSuccessHandler((res) => {
            btn.disabled = false;
            if(spinner) spinner.style.display = 'none';
            cerrarModal();
            mostrarMensaje("✅ " + res, "exito");
          })
          .withFailureHandler((err) => {
            btn.disabled = false;
            if(spinner) spinner.style.display = 'none';
            mostrarMensaje("❌ Error: " + err.message, "error");
          })
          .iniciarFormularioSeguimiento();
      }
