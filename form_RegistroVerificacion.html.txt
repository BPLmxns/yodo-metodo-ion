  <script>
    window.onload = () => {
      const ahora = new Date();
      const anio = ahora.getFullYear();
      const mes = String(ahora.getMonth() + 1).padStart(2, '0');
      const dia = String(ahora.getDate()).padStart(2, '0');
      const horas = String(ahora.getHours()).padStart(2, '0');
      const minutos = String(ahora.getMinutes()).padStart(2, '0');
      
      document.getElementById('fechaHora').value = `${anio}-${mes}-${dia} - ${horas}:${minutos}`;

      google.script.run
        .withSuccessHandler(poblarListasDesplegables)
        .withFailureHandler(err => {
          document.getElementById('loader').style.display = 'none';
          mostrarToastError(err.message);
        })
        .obtenerLotesActivos();

      asignarEventosValidacion();
    };

    function poblarListasDesplegables(lotes) {
      const selectYoduro = document.getElementById('loteYoduro');
      const selectIsa = document.getElementById('loteIsa');
      
      selectYoduro.innerHTML = '<option value="">Seleccione un lote...</option>';
      selectIsa.innerHTML = '<option value="">Seleccione un lote...</option>';
      
      if (lotes.yoduro.length === 0) {
        selectYoduro.innerHTML += '<option value="" disabled>No hay lotes vigentes</option>';
      } else {
        lotes.yoduro.forEach(lote => { selectYoduro.innerHTML += `<option value="${lote}">${lote}</option>`; });
      }
      
      if (lotes.isa.length === 0) {
        selectIsa.innerHTML += '<option value="" disabled>No hay lotes vigentes</option>';
      } else {
        lotes.isa.forEach(lote => { selectIsa.innerHTML += `<option value="${lote}">${lote}</option>`; });
      }
      
      document.getElementById('loader').style.display = 'none';
    }

    function calcularDiferenciaYConformidad() {
      const val1 = parseFloat(document.getElementById('resp1').value);
      const val10 = parseFloat(document.getElementById('resp10').value);
      const campoDiferencia = document.getElementById('diferencia');
      const badge = document.getElementById('statusBadge');

      if (!isNaN(val1) && !isNaN(val10)) {
        const diferencia = Math.abs(val10 - val1);
        campoDiferencia.value = diferencia.toFixed(1) + " mV";

        if (diferencia >= 54.0 && diferencia <= 60.0) {
          badge.innerText = "CONFORME";
          badge.className = "badge badge-conforme";
        } else {
          badge.innerText = "NO CONFORME";
          badge.className = "badge badge-noconforme";
        }
      } else {
        campoDiferencia.value = "";
        badge.innerText = "Pendiente de ingreso de mV";
        badge.className = "badge";
      }
    }

    function mostrarToastError(mensaje) {
      const container = document.getElementById('toast-container');
      const toast = document.createElement('div');
      toast.className = `toast toast-error`;
      toast.innerHTML = `⚠️ ${mensaje}`;
      container.appendChild(toast);
      setTimeout(() => toast.classList.add('show'), 10);
      setTimeout(() => { toast.classList.remove('show'); setTimeout(() => toast.remove(), 300); }, 3500);
    }

    function asignarEventosValidacion() {
      const inputs = document.querySelectorAll('input:not(.read-only), select');
      inputs.forEach(input => {
        input.addEventListener('input', function() {
          if (this.value.trim() !== "") this.classList.remove('invalid');
        });
      });
    }

    function intentarRegistrar() {
      const loteYod = document.getElementById('loteYoduro');
      const loteIs = document.getElementById('loteIsa');
      const r1 = document.getElementById('resp1');
      const r10 = document.getElementById('resp10');
      const badge = document.getElementById('statusBadge');

      let hayErrores = false;

      if (!loteYod.value) { loteYod.classList.add('invalid'); hayErrores = true; }
      if (!loteIs.value) { loteIs.classList.add('invalid'); hayErrores = true; }
      if (r1.value === "") { r1.classList.add('invalid'); hayErrores = true; }
      if (r10.value === "") { r10.classList.add('invalid'); hayErrores = true; }

      if (hayErrores) {
        mostrarToastError("Todos los campos del formulario son de diligenciamiento obligatorio");
        return;
      }

      const val1 = parseFloat(r1.value);
      const val10 = parseFloat(r10.value);
      const diff = Math.abs(val10 - val1);

      const datos = {
        loteYoduro: loteYod.value,
        loteIsa: loteIs.value,
        resp1: val1,
        resp10: val10,
        diferencia: diff,
        conformidad: badge.innerText
      };

      // MOSTRAR PANTALLA DE CARGA CENTRAL
      document.getElementById('overlay-submit').style.display = 'flex';
      document.getElementById('submit-spinner').style.display = 'block';
      document.getElementById('submit-icon-success').style.display = 'none';
      document.getElementById('submit-title').innerText = 'Registrando verificación...';
      document.getElementById('submit-title').style.color = 'var(--azul-oscuro)';
      document.getElementById('submit-text').innerText = 'Por favor, no cierre la ventana.';

      google.script.run
        .withSuccessHandler((res) => {
          // TRANSICIÓN AL MENSAJE DE ÉXITO
          document.getElementById('submit-spinner').style.display = 'none';
          document.getElementById('submit-icon-success').style.display = 'block';
          
          document.getElementById('submit-title').innerText = '¡Proceso Completado!';
          document.getElementById('submit-title').style.color = 'var(--success)';
          document.getElementById('submit-text').innerText = res + '\nCerrando ventana...';
          
          setTimeout(() => { google.script.host.close(); }, 2500);
        })
        .withFailureHandler((err) => {
          // OCULTAR PANTALLA Y MOSTRAR TOAST DE ERROR
          document.getElementById('overlay-submit').style.display = 'none';
          mostrarToastError(err.message);
        })
        .guardarVerificacion(datos);
    }
  </script>
